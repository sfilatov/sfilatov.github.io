--- layout: post title: RubyMine launcher on Ubuntu 11.10 Unity date: '2011-10-17T02:12:00.000-07:00' author: "Степан" tags: modified_time: '2011-10-17T02:18:13.890-07:00' blogger_id: tag:blogger.com,1999:blog-6563657311735677139.post-4448247786807725332 blogger_orig_url: http://sfilatov.blogspot.com/2011/10/rubymine-launcher-on-ubuntu-1110-unity.html --- Create file ~/.rubymine.sh

#rubymine.sh content
export JAVA_HOME=/usr/lib/jvm/java-6-sun
export JDK_HOME=$JAVA_HOME
export RUBYMINE_JDK=$JAVA_HOME
/%rubymine folder%/bin/rubymine.sh

Create RubyMine.desktop file in ~/.local/share/applications/

#RubyMine.desktop
[Desktop Entry]
Name=RubyMine
Exec=/%user home%/rubymine.sh
Icon=/%rubymine folder%/bin/rubymine.png
Terminal=false
Type=Application
StartupNotify=true

Drag and Drop RubyMine.desktop to Ubuntu launcher
